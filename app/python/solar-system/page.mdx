# Cartoon Solar System Adventure

<Callout tone="note">This playful mission introduces the Python <code>turtle</code> module while sneaking in friendly doses of geometry, loops, and animation. Each stop on the journey adds a new visual so learners always see what their code does.</Callout>

## What you will build
- A star-sprinkled night sky as the stage for our cartoon universe
- A bright sun plus colorful planets and moons sized by simple math
- Orbit paths drawn with loops so planets travel in smooth circles
- An optional animation loop that makes everything revolve in real time

## Mission prep: tools and theory

### 1. Install and test turtle
Python includes <code>turtle</code> by default on Windows and macOS. Open a terminal and run:

```bash
python --version
python - <<'PY'
import turtle
print('turtle ready for launch!')
PY
```

If a window pops up (even briefly), the module is ready.

### 2. Quick theory pit stop
- The turtle canvas is a 2D coordinate plane with <code>(0, 0)</code> in the center.
- Angles turn counter-clockwise; one lap is 360 degrees.
- Circles are drawn by asking the turtle to walk and turn a tiny bit repeatedly.
- Orbit speed feels right when distance traveled each frame stays proportional to the orbit radius (think circumference: <code>2 * pi * r</code>).

## Full mission code

Put everything together in one file named <code>cartoon_solar_system.py</code> so learners can run it end to end.

```python
import math
import random
import turtle as t

screen = t.Screen()
screen.setup(width=800, height=800)
screen.bgcolor("midnight blue")
screen.title("Cartoon Solar System")

pen = t.Turtle()
pen.hideturtle()
pen.speed(0)

# stars
def draw_star(x, y):
    pen.up()
    pen.goto(x, y)
    pen.down()
    pen.begin_fill()
    for _ in range(5):
        pen.forward(10)
        pen.right(144)
    pen.end_fill()

pen.color("white")
for _ in range(40):
    draw_star(random.randint(-380, 380), random.randint(-360, 360))

# sun
sun = t.Turtle()
sun.hideturtle()
sun.speed(0)
sun.up()
sun.goto(0, -80)
sun.down()
sun.color("gold", "goldenrod")
sun.begin_fill()
sun.circle(80)
sun.end_fill()
sun.up()
sun.color("orange")
sun.goto(0, -110)
sun.down()
sun.pensize(4)
sun.circle(110)

# orbit guide
pen.pensize(1)
pen.color("gray25")

planet_specs = [
    (140, 15, "light sky blue", 2.5, "Orbit of Aqua"),
    (200, 20, "pale violet red", 1.8, "Orbit of Berry"),
    (270, 30, "medium sea green", 1.2, "Orbit of Lime"),
]

for radius, _, _, _, label in planet_specs:
    pen.up()
    pen.goto(0, -radius)
    pen.down()
    pen.circle(radius)
    pen.up()
    pen.goto(0, -radius - 40)
    pen.color("white")
    pen.write(label, align="center", font=("Arial", 10, "bold"))
    pen.color("gray25")

screen.tracer(0)

class Planet:
    def __init__(self, radius, size, color, speed):
        self.radius = radius
        self.angle = random.randint(0, 359)
        self.speed = speed
        turtle = t.Turtle()
        turtle.hideturtle()
        turtle.speed(0)
        turtle.shape("circle")
        turtle.color(color)
        turtle.shapesize(size / 10)
        turtle.up()
        self.turtle = turtle

    def move(self):
        self.angle = (self.angle + self.speed) % 360
        radians = math.radians(self.angle)
        x = self.radius * math.cos(radians)
        y = self.radius * math.sin(radians)
        self.turtle.goto(x, y)

planets = [
    Planet(radius, size, color, speed)
    for radius, size, color, speed, _ in planet_specs
]

class Moon(Planet):
    def __init__(self, host, radius, size, color, speed):
        super().__init__(radius, size, color, speed)
        self.host = host

    def move(self):
        self.angle = (self.angle + self.speed) % 360
        radians = math.radians(self.angle)
        host_x, host_y = self.host.turtle.position()
        x = host_x + self.radius * math.cos(radians)
        y = host_y + self.radius * math.sin(radians)
        self.turtle.goto(x, y)

moon = Moon(host=planets[-1], radius=40, size=8, color="light gray", speed=4)
planets.append(moon)

comet = t.Turtle(shape="triangle")
comet.color("orange red")
comet.shapesize(stretch_wid=0.5, stretch_len=2.5)
comet.up()
angle = 0

while True:
    for planet in planets:
        planet.move()
    comet.setheading(angle)
    comet.goto(-360 + angle * 2, 240 - angle * 0.8)
    angle = (angle + 1) % 360
    screen.update()

# keep window open if loop is ever exited
screen.mainloop()
```
