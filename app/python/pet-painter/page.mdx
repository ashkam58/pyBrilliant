<Step num={6} title="Draw simple cartoon pets" minutes={15}>
We‚Äôll make three functions: `pet_cat`, `pet_fish`, and `pet_turtle`. They use circles and triangles to suggest each animal. Not Pixar, but undeniably cute.

```python
# Simple helper shapes (filled)

def filled_circle(r):
    pen.begin_fill()
    pen.circle(r)
    pen.end_fill()


def filled_triangle(side):
    pen.begin_fill()
    for _ in range(3):
        pen.forward(side)
        pen.left(120)
    pen.end_fill()


def pet_cat():
    set_random_color(); filled_circle(60)  # face
    # ears
    set_random_color(); pen.up(); pen.goto(-40, 60); pen.setheading(0); pen.down(); filled_triangle(30)
    set_random_color(); pen.up(); pen.goto(40, 60); pen.setheading(0); pen.down(); filled_triangle(30)
    # eyes
    for x in (-20, 20):
        pen.up(); pen.goto(x, 20); pen.down()
        pen.color("white"); pen.begin_fill(); pen.circle(8); pen.end_fill()
        pen.color("black"); pen.begin_fill(); pen.circle(3); pen.end_fill()
    # nose (tiny triangle)
    pen.up(); pen.goto(0, 5); pen.down();
    for _ in range(3):
        pen.forward(10); pen.left(120)


def pet_fish():
    set_random_color(); pen.begin_fill(); pen.circle(80, 180); pen.circle(80, -180); pen.end_fill()  # body-ish
    # tail
    set_random_color(); pen.up(); pen.goto(80, 0); pen.setheading(0); pen.down()
    for _ in range(2):
        pen.begin_fill(); pen.forward(30); pen.left(120); pen.forward(30); pen.left(60); pen.end_fill(); pen.left(180)
    # eye
    pen.up(); pen.goto(-30, 10); pen.down(); pen.color("white"); pen.begin_fill(); pen.circle(10); pen.end_fill()
    pen.color("black"); pen.begin_fill(); pen.circle(4); pen.end_fill()


def pet_turtle():
    set_random_color(); pen.begin_fill(); pen.circle(60); pen.end_fill()  # shell
    # head
    pen.up(); pen.goto(60, 0); pen.setheading(0); pen.down(); set_random_color(); filled_circle(20)
    # legs
    for pos in [(-40, -40), (-40, 40), (40, -40), (40, 40)]:
        pen.up(); pen.goto(*pos); pen.down(); set_random_color(); filled_circle(15)
```

Glue the names to functions so we can call them neatly:

```python
pet_map = {
    "cat": pet_cat,
    "fish": pet_fish,
    "turtle": pet_turtle,
}

pet_fn = pet_map.get(choice, pet_turtle)  # default to turtle if unknown
```

<Step num={7} title="Events: keys & confetti party" minutes={8}>
Time to celebrate. We‚Äôll add a `confetti(n)` function and some keyboard shortcuts.

```python
import random  # (If you didn't already)

def confetti(n=80):
    pen.up()
    for _ in range(n):
        pen.goto(random.randint(-230, 230), random.randint(-170, 170))
        pen.dot(random.randint(5, 15), random.choice(colors))
    pen.down()

# Draw the chosen pet at a cozy spot
pen.up(); pen.goto(0, -60); pen.down()
pet_fn()

# Write a label
pen.up(); pen.goto(0, -120); pen.down(); pen.color("black")
pen.write(f"This is {choice.title()}!", align="center", font=("Arial", 18, "bold"))

# Add keys: c = confetti, r = redraw

def redraw():
    pen.clear()
    pen.up(); pen.goto(0, -60); pen.down()
    pet_fn()
    pen.up(); pen.goto(0, -120); pen.down(); pen.color("black")
    pen.write(f"This is {choice.title()}!", align="center", font=("Arial", 18, "bold"))

screen.listen()
screen.onkey(lambda: confetti(100), "c")
screen.onkey(redraw, "r")
```

<Step num={8} title="Polish & full program" minutes={10}>
Here‚Äôs a clean version you can copy-paste into `pet_painter.py` if you want everything in one place:

```python
import turtle as t
import random

screen = t.Screen()
screen.title("Pet Painter üê¢üé®")
screen.bgcolor("#fcefe3")

pen = t.Turtle()
pen.shape("turtle")
pen.pensize(3)
pen.speed(0)

colors = ["#ff6b6b", "#ffd93d", "#6bcb77", "#4d96ff", "#c77dff", "#ffb5e8", "#40e0d0"]

# Helpers

def set_random_color():
    pen.color(random.choice(colors))

def filled_circle(r):
    pen.begin_fill(); pen.circle(r); pen.end_fill()

def filled_triangle(side):
    pen.begin_fill()
    for _ in range(3):
        pen.forward(side); pen.left(120)
    pen.end_fill()

# Pets

def pet_cat():
    set_random_color(); filled_circle(60)  # face
    set_random_color(); pen.up(); pen.goto(-40, 60); pen.setheading(0); pen.down(); filled_triangle(30)
    set_random_color(); pen.up(); pen.goto(40, 60); pen.setheading(0); pen.down(); filled_triangle(30)
    for x in (-20, 20):
        pen.up(); pen.goto(x, 20); pen.down()
        pen.color("white"); pen.begin_fill(); pen.circle(8); pen.end_fill()
        pen.color("black"); pen.begin_fill(); pen.circle(3); pen.end_fill()
    pen.up(); pen.goto(0, 5); pen.down()
    for _ in range(3):
        pen.forward(10); pen.left(120)


def pet_fish():
    set_random_color(); pen.begin_fill(); pen.circle(80, 180); pen.circle(80, -180); pen.end_fill()
    set_random_color(); pen.up(); pen.goto(80, 0); pen.setheading(0); pen.down()
    for _ in range(2):
        pen.begin_fill(); pen.forward(30); pen.left(120); pen.forward(30); pen.left(60); pen.end_fill(); pen.left(180)
    pen.up(); pen.goto(-30, 10); pen.down(); pen.color("white"); pen.begin_fill(); pen.circle(10); pen.end_fill()
    pen.color("black"); pen.begin_fill(); pen.circle(4); pen.end_fill()


def pet_turtle():
    set_random_color(); pen.begin_fill(); pen.circle(60); pen.end_fill()
    pen.up(); pen.goto(60, 0); pen.setheading(0); pen.down(); set_random_color(); filled_circle(20)
    for pos in [(-40, -40), (-40, 40), (40, -40), (40, 40)]:
        pen.up(); pen.goto(*pos); pen.down(); set_random_color(); filled_circle(15)

pet_map = {"cat": pet_cat, "fish": pet_fish, "turtle": pet_turtle}

# Ask the user
choice = screen.textinput("Pet Painter", "Pick a pet: cat, fish, or turtle")
if not choice:
    choice = input("Pick a pet: cat, fish, or turtle: ")
choice = choice.strip().lower()
pet_fn = pet_map.get(choice, pet_turtle)

# Confetti!

def confetti(n=100):
    pen.up()
    for _ in range(n):
        pen.goto(random.randint(-230, 230), random.randint(-170, 170))
        pen.dot(random.randint(5, 15), random.choice(colors))
    pen.down()

# Draw + label
pen.up(); pen.goto(0, -60); pen.down(); pet_fn()
pen.up(); pen.goto(0, -120); pen.down(); pen.color("black")
pen.write(f"This is {choice.title()}!", align="center", font=("Arial", 18, "bold"))

# Keys

def redraw():
    pen.clear()
    pen.up(); pen.goto(0, -60); pen.down(); pet_fn()
    pen.up(); pen.goto(0, -120); pen.down(); pen.color("black")
    pen.write(f"This is {choice.title()}!", align="center", font=("Arial", 18, "bold"))

screen.listen()
screen.onkey(lambda: confetti(100), "c")
screen.onkey(redraw, "r")

screen.mainloop()
```

## Vocabulary & Memory Tricks

- **Variable**: a labeled box that remembers a value.  
- **Function**: a named recipe you can reuse.  
- **Loop**: do it again (politely, a set number of times).  
- **Event**: something happens (like a key press) ‚Üí run a function.  
- **Random**: ask the universe to surprise you.

**Mnemonic**: *Very Fancy Llamas Eat Ravioli* ‚Üí **V**ariables, **F**unctions, **L**oops, **E**vents, **R**andom.

## Bug Clinic

- ‚Äú**NameError: set_random_color is not defined**‚Äù ‚Üí The function must be defined before you call it (or at least in the same file).
- ‚Äú**IndentationError**‚Äù ‚Üí Python cares about spacing. Make blocks line up consistently.
- Nothing draws? Make sure you created the `screen` and `pen`, and didn‚Äôt forget `screen.mainloop()`.
