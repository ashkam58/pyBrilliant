### Challenge 1: Password Validator 🔐

Create a method that validates passwords based on these criteria:
- At least 8 characters long
- Contains at least one uppercase letter
- Contains at least one lowercase letter  
- Contains at least one digit

<JavaCodeRunner
  code={`public class PasswordValidator {
    
    public static boolean isValidPassword(String password) {
        // Check length
        if (password.length() < 8) {
            return false;
        }
        
        boolean hasUpper = false;
        boolean hasLower = false;
        boolean hasDigit = false;
        
        // Check each character
        for (int i = 0; i < password.length(); i++) {
            char c = password.charAt(i);
            
            if (Character.isUpperCase(c)) {
                hasUpper = true;
            } else if (Character.isLowerCase(c)) {
                hasLower = true;
            } else if (Character.isDigit(c)) {
                hasDigit = true;
            }
        }
        
        return hasUpper && hasLower && hasDigit;
    }
    
    public static void testPassword(String password) {
        boolean valid = isValidPassword(password);
        System.out.println("Password: " + password);
        System.out.println("Valid: " + (valid ? "✅ Yes" : "❌ No"));
        System.out.println();
    }
    
    public static void main(String[] args) {
        System.out.println("🔐 Password Validator Test");
        System.out.println("=" .repeat(30));
        
        // Test different passwords
        testPassword("password");        // Too simple
        testPassword("Password");        // Missing digit
        testPassword("password123");     // Missing uppercase
        testPassword("PASSWORD123");     // Missing lowercase
        testPassword("Password123");     // Valid!
        testPassword("MySecure1");       // Valid!
        testPassword("short1A");         // Too short
    }
}`}

<JavaCodeRunner

class Student {
    private String name;

### Challenge 3: Number Pattern Generator 🎨

Create a program that generates various number patterns:

<JavaCodeRunner
  code={`public class PatternGenerator {
    
    public static void printNumberTriangle(int rows) {
        System.out.println("🔺 Number Triangle (" + rows + " rows):");
        for (int i = 1; i <= rows; i++) {
            // Print spaces for alignment
            for (int j = 1; j <= rows - i; j++) {
                System.out.print(" ");
            }
            // Print numbers
            for (int j = 1; j <= i; j++) {
                System.out.print(j + " ");
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void printMultiplicationTable(int size) {
        System.out.println("🔢 Multiplication Table (" + size + "x" + size + "):");
        
        // Print header
        System.out.print("   ");
        for (int i = 1; i <= size; i++) {
            System.out.printf("%4d", i);
        }
        System.out.println();
        System.out.println("   " + "-".repeat(size * 4));
        
        // Print table
        for (int i = 1; i <= size; i++) {
            System.out.printf("%2d|", i);
            for (int j = 1; j <= size; j++) {
                System.out.printf("%4d", i * j);
            }
            System.out.println();
        }
        System.out.println();
    }
    
    public static void printFibonacci(int terms) {
        System.out.println("🌀 Fibonacci Sequence (" + terms + " terms):");
        
        if (terms <= 0) {
            System.out.println("Invalid number of terms!");
            return;
        }
        
        long first = 0, second = 1;
        
        if (terms >= 1) System.out.print(first + " ");
        if (terms >= 2) System.out.print(second + " ");
        
        for (int i = 3; i <= terms; i++) {
            long next = first + second;
            System.out.print(next + " ");
            first = second;
            second = next;
        }
        System.out.println("\\n");
    }
    
    public static void printPrimeNumbers(int limit) {
        System.out.println("✨ Prime Numbers up to " + limit + ":");
        
        for (int num = 2; num <= limit; num++) {
            if (isPrime(num)) {
                System.out.print(num + " ");
            }
        }
        System.out.println("\\n");
    }
    
    private static boolean isPrime(int number) {
        if (number <= 1) return false;
        if (number <= 3) return true;
        if (number % 2 == 0 || number % 3 == 0) return false;
        
        for (int i = 5; i * i <= number; i += 6) {
            if (number % i == 0 || number % (i + 2) == 0) {
                return false;
            }
        }
        return true;
    }
    
    public static void main(String[] args) {
        System.out.println("🎨 PATTERN GENERATOR");
        System.out.println("=" .repeat(40));
        
        printNumberTriangle(5);
        printMultiplicationTable(5);
        printFibonacci(10);
        printPrimeNumbers(30);
        
        System.out.println("🎉 Pattern generation complete!");
    }
}`}

## Part 4: Final Project - Mini Bank System 🏦

Your capstone project: Create a complete banking system with multiple account types!

<JavaCodeRunner
  code={`import java.util.ArrayList;

abstract class Account {
    protected String accountNumber;
    protected String ownerName;
    protected double balance;
    protected static int nextAccountNumber = 1000;
    
    public Account(String ownerName, double initialBalance) {
        this.accountNumber = "ACC" + (++nextAccountNumber);
        this.ownerName = ownerName;
        this.balance = Math.max(0, initialBalance);
    }
    
    public abstract void deposit(double amount);
    public abstract boolean withdraw(double amount);
    
    public double getBalance() { return balance; }
    public String getAccountNumber() { return accountNumber; }
    public String getOwnerName() { return ownerName; }
    
    public void displayInfo() {
        System.out.println("Account: " + accountNumber);
        System.out.println("Owner: " + ownerName);
        System.out.println("Balance: $" + String.format("%.2f", balance));
    }
}

class SavingsAccount extends Account {
    private double interestRate;
    private int withdrawalCount;
    private static final int MAX_WITHDRAWALS = 3;
    
    public SavingsAccount(String ownerName, double initialBalance, double interestRate) {
        super(ownerName, initialBalance);
        this.interestRate = interestRate;
        this.withdrawalCount = 0;
    }
    
    @Override
    public void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            System.out.println("💰 Deposited $" + amount + " to savings account");
        } else {
            System.out.println("❌ Invalid deposit amount");
        }
    }
    
    @Override
    public boolean withdraw(double amount) {
        if (withdrawalCount >= MAX_WITHDRAWALS) {
            System.out.println("❌ Maximum withdrawals exceeded for this period");
            return false;
        }
        
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            withdrawalCount++;
            System.out.println("💸 Withdrew $" + amount + " from savings account");
            System.out.println("Withdrawals remaining: " + (MAX_WITHDRAWALS - withdrawalCount));
            return true;
        } else {
            System.out.println("❌ Invalid withdrawal amount or insufficient funds");
            return false;
        }
    }
    
    public void addInterest() {
        double interest = balance * (interestRate / 100);
        balance += interest;
        System.out.println("📈 Interest added: $" + String.format("%.2f", interest));
    }
    
    @Override
    public void displayInfo() {
        System.out.println("💰 SAVINGS ACCOUNT");
        super.displayInfo();
        System.out.println("Interest Rate: " + interestRate + "%");
        System.out.println("Withdrawals Used: " + withdrawalCount + "/" + MAX_WITHDRAWALS);
    }
}

class CheckingAccount extends Account {
    private double overdraftLimit;
    
    public CheckingAccount(String ownerName, double initialBalance, double overdraftLimit) {
        super(ownerName, initialBalance);
        this.overdraftLimit = overdraftLimit;
    }
    
    @Override
    public void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            System.out.println("💰 Deposited $" + amount + " to checking account");
        } else {
            System.out.println("❌ Invalid deposit amount");
        }
    }
    
    @Override
    public boolean withdraw(double amount) {
        if (amount > 0 && (balance + overdraftLimit) >= amount) {
            balance -= amount;
            System.out.println("💸 Withdrew $" + amount + " from checking account");
            
            if (balance < 0) {
                System.out.println("⚠️ Account overdrawn! Balance: $" + String.format("%.2f", balance));
            }
            return true;
        } else {
            System.out.println("❌ Withdrawal exceeds overdraft limit");
            return false;
        }
    }
    
    @Override
    public void displayInfo() {
        System.out.println("💳 CHECKING ACCOUNT");
        super.displayInfo();
        System.out.println("Overdraft Limit: $" + String.format("%.2f", overdraftLimit));
        
        if (balance < 0) {
            System.out.println("⚠️ Account is overdrawn!");
        }
    }
}

class Bank {
    private ArrayList<Account> accounts;
    private String bankName;
    
    public Bank(String bankName) {
        this.bankName = bankName;
        this.accounts = new ArrayList<>();
    }
    
    public void createSavingsAccount(String ownerName, double initialBalance, double interestRate) {
        SavingsAccount account = new SavingsAccount(ownerName, initialBalance, interestRate);
        accounts.add(account);
        System.out.println("✅ Created savings account: " + account.getAccountNumber());
    }
    
    public void createCheckingAccount(String ownerName, double initialBalance, double overdraftLimit) {
        CheckingAccount account = new CheckingAccount(ownerName, initialBalance, overdraftLimit);
        accounts.add(account);
        System.out.println("✅ Created checking account: " + account.getAccountNumber());
    }
    
    public Account findAccount(String accountNumber) {
        for (Account account : accounts) {
            if (account.getAccountNumber().equals(accountNumber)) {
                return account;
            }
        }
        return null;
    }
    
    public void displayAllAccounts() {
        System.out.println("\\n🏦 " + bankName + " - All Accounts");
        System.out.println("=" .repeat(50));
        
        if (accounts.isEmpty()) {
            System.out.println("No accounts found.");
            return;
        }
        
        double totalBalance = 0;
        for (Account account : accounts) {
            account.displayInfo();
            totalBalance += account.getBalance();
            System.out.println();
        }
        
        System.out.println("💰 Total Bank Assets: $" + String.format("%.2f", totalBalance));
    }
}

public class BankingSystem {
    public static void main(String[] args) {
        Bank bank = new Bank("Java National Bank");
        
        System.out.println("🏦 Welcome to Java National Bank!");
        System.out.println("=" .repeat(40));
        
        // Create accounts
        bank.createSavingsAccount("Alice Johnson", 1500.0, 2.5);
        bank.createCheckingAccount("Bob Smith", 800.0, 500.0);
        bank.createSavingsAccount("Charlie Brown", 2200.0, 3.0);
        
        System.out.println();
        
        // Perform transactions
        Account aliceAccount = bank.findAccount("ACC1001");
        if (aliceAccount != null) {
            aliceAccount.deposit(200);
            aliceAccount.withdraw(100);
            
            if (aliceAccount instanceof SavingsAccount) {
                ((SavingsAccount) aliceAccount).addInterest();
            }
        }
        
        System.out.println();
        
        Account bobAccount = bank.findAccount("ACC1002");
        if (bobAccount != null) {
            bobAccount.withdraw(1200); // This will use overdraft
            bobAccount.deposit(300);
        }
        
        // Display final state
        bank.displayAllAccounts();
        
        System.out.println("🎉 Banking system demo complete!");
    }
}`}
